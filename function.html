C Program to show different types of Input/Output Functions

Code

Formatted Input/Output Functions (printf and scanf) 

      #include < stdio.h >
int main()
{
	int a;
// display output
	printf("Enter your favourite number: ");
// to get input
	scanf("%d",&a);
	printf("Your favourite number is %d",a);
}

Unformatted Input/Output Functions

putchar() 

        #include < stdio.h >
	int main()
	{
 
    	// the character to be written
   	 char ch = 'G';
 
   	 // to display character as output
   	 putchar(ch);
 
   	 return (0);
	}

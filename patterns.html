C Program Demonstrating Different Patterns

Code

#include <stdio.h>

int main() {
    int n = 5;

    // Square Pattern
    printf("Square Pattern:\n");
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            printf("* ");
        }
        printf("\n");
    }
    printf("\n");

    // Rectangle Pattern
    printf("Rectangle Pattern:\n");
    int rows = 3, cols = 7;
    for (int i = 0; i < rows; i++) {
        for (int j = 0; j < cols; j++) {
            printf("* ");
        }
        printf("\n");
    }
    printf("\n");

    // Right Pyramid Pattern
    printf("Right Pyramid Pattern:\n");
    for (int i = 1; i <= n; i++) {
        for (int j = 0; j < i; j++) {
            printf("* ");
        }
        printf("\n");
    }
    printf("\n");

    // Inverse Right Pyramid Pattern
    printf("Inverse Right Pyramid Pattern:\n");
    for (int i = n; i > 0; i--) {
        for (int j = 0; j < i; j++) {
            printf("* ");
        }
        printf("\n");
    }

    return 0;
}

                #include < stdio.h >
                    #include < string.h >
                    
                    int main() {
                        char str1[50] = "Hello";
                        char str2[50] = "World";
                        char str3[50];
                        char *pos;
                    
                        // strlen
                        printf("Length of str1: %lu\n", strlen(str1));
                    
                        // strcpy
                        strcpy(str3, str1);
                        printf("str3 after strcpy: %s\n", str3);
                    
                        // strcat
                        strcat(str1, str2);
                        printf("str1 after strcat: %s\n", str1);
                    
                        // strcmp
                        int cmp = strcmp(str1, str2);
                        printf("strcmp result: %d\n", cmp);
                    
                        // strrev
                        strrev(str1);
                        printf("str1 after strrev: %s\n", str1);
                    
                        // strlwr
                        strlwr(str1);
                        printf("str1 after strlwr: %s\n", str1);
                    
                        // strupr
                        strupr(str1);
                        printf("str1 after strupr: %s\n", str1);
                        
                        return 0;
